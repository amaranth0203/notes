print_r 用来打印整个数组的结构
var_dump 也可以打印整个数组的结构，并且附加键值对的类型

foreach( $arr as $value )  - 传值
foreach( $arr as &$value ) - 传址

foreach( $arr as $value ) { }
while( list( , $value ) = each( $arr ) ) { }

foreach( $arr as $key => $value ) { }
while( list( $key , $value ) = each( $arr ) ) { }

break 2 ; 跳出两层
